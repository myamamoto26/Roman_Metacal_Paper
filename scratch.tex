    \vb*{M_{ij}} = 
    \frac{\langle e^{+}_{i}/R_{ii} \rangle - \langle e^{-}_{i}/R_{ii} \rangle}{\abs{\Delta g_{j}}} - \delta_{ij}

\end{equation}
